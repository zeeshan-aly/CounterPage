    }

      color: red;
      display: none; /* Initially hide the error message */
    }

      display: flex;

<body>
  <div class="container">
    <div class="heading">Counter using HTML</div>
    <div class="counter">Your current count is: <span id="count">0</span></div>
    <div class="error" id="error-message">Count cannot go below 0</div>
    <div class="button-container">
      <button id="decrement">Decrement</button>
      <button id="increment">Increment</button>
      <button id="clear">Clear</button>
    </div>
  </div>

  <script>
    const countElement = document.getElementById('count');
    const incrementBtn = document.getElementById('increment');
    const decrementBtn = document.getElementById('decrement');
    const clearBtn = document.getElementById('clear');
    const errorMessage = document.getElementById('error-message');

    let count = 0;

    function updateCounter() {
      countElement.textContent = count;
      if (count === 0) {
        errorMessage.style.display = 'block'; // Show error message if count is zero
        decrementBtn.classList.add('fade-out'); // Fade out decrement button
        decrementBtn.disabled = true;
      } else {
        errorMessage.style.display = 'none'; // Hide error message if count is non-zero
        decrementBtn.classList.remove('fade-out'); // Remove fade-out from decrement button
        decrementBtn.disabled = false;
      }
    }

    incrementBtn.addEventListener('click', () => {
      count++;
      updateCounter();
    });

    decrementBtn.addEventListener('click', () => {
      if (count > 0) {
        count--;
      }
      updateCounter();
    });

    clearBtn.addEventListener('click', () => {
      count = 0;
      updateCounter();
    });

    updateCounter();
  </script>
